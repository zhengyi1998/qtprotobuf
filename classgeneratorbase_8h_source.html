93  template<typename T>
94  void printQEnums(const T *message) {
95  if (message->enum_type_count() <= 0) {
96  return;
97  }
98 
99  printPublic();
100 
101  Indent();
102  for (int i = 0; i < message->enum_type_count(); i++) {
103  const auto enumDescr = message->enum_type(i);
104  mPrinter->Print({{"enum", enumDescr->name()}}, Templates::EnumDefinitionTemplate);
105 
106  Indent();
107  for (int j = 0; j < enumDescr->value_count(); j++) {
108  const auto valueDescr = enumDescr->value(j);
109  mPrinter->Print({{"enumvalue", utils::upperCaseName(valueDescr->name())},
110  {"value", std::to_string(valueDescr->number())}}, Templates::EnumFieldTemplate);
111  }
112  Outdent();
113  mPrinter->Print(Templates::SemicolonBlockEnclosureTemplate);
114  mPrinter->Print({{"type", enumDescr->name().c_str()}}, Templates::QEnumTemplate);
115  }
116 
117  for (int i = 0; i < message->enum_type_count(); i++) {
118  const auto enumDescr = message->enum_type(i);
119  mPrinter->Print({{"enum", enumDescr->name()}}, Templates::EnumTypeRepeatedTemplate);
120  }
121  Outdent();
122  }

124  template<typename T>
125  void printComments(T *descriptor) {
126  if (!GeneratorOptions::instance().generateComments()) {
127  return;
128  }
129 
130  ::google::protobuf::SourceLocation loc;
131  descriptor->GetSourceLocation(&loc);
132 
133  utils::trim(loc.leading_comments);
134  if (loc.leading_comments.size() > 0) {
135  auto firstEntry = loc.leading_comments.find('\n');
136  bool isSingleLine = firstEntry == std::string::npos;
137 
138  if (loc.leading_comments[0] != '!' && loc.leading_comments[0] != '*' && loc.leading_comments[0] != ' ') {
139  loc.leading_comments = " " + loc.leading_comments;
140  if (!isSingleLine) {
141  loc.leading_comments = "\n" + loc.leading_comments;
142  }
143  }
144  mPrinter->Print("\n/*");
145  if (isSingleLine) {
146  mPrinter->Print(loc.leading_comments.c_str());
147  } else {
148  utils::replace(loc.leading_comments, "\n", "\n *");
149  mPrinter->Print(loc.leading_comments.c_str());
150  if (loc.leading_comments[loc.leading_comments.size() - 1] != '\n') {
151  mPrinter->Print("\n");
152  }
153  }
154  mPrinter->Print(" */");
155  }
156  }
